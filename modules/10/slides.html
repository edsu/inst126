Add to a Set

nle = set([
    "Braves",
    "Marlins",
    "Mets",
    "Phillies"
])

nle.add("Nationals")

{"Braves", "Marlins", "Mets", Phillies", "Nationals"}

Remove from a Set

    "Braves",

Union

a = set([1, 2, 3])
b = set([4, 5, 6])

print(a.union(b))
{1, 2, 3, 4, 5, 6}

Intersection

a = set([1, 2, 3, 4])
b = set([3, 4, 5, 6])

print(a.intersection(b))
{3, 4}

Difference

a = set([1, 2, 3, 4])
b = set([3, 4, 5, 6])

print(a.difference(b))
{1, 2}

Subset

a = set([1, 2, 3])
b = set([1, 2, 3, 4, 5, 6])

print(a.issubset(b))
True
print(b.issubset(a))
False
